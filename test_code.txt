def encrypt(text,s):
    #enter text that will be converted
    #enter s which means the number of shifts that will be made to 
    #change the letter
    result = ""
    
    #traverse text
    for i in range(len(text)):
        char = text[i]
    
        #encrypt uppercase characters
        if char.isupper():
            result += chr((ord(char) + s - 65) % 26 + 65)
        
        #encrypt lowercase characters
        else:
            result += chr((ord(char) + s - 97) % 26 + 97)
    
    return result
